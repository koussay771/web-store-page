// src/Chatbot.jsx
import React, { useState, useEffect, useRef } from 'react';

const Chatbot = () => {
    // State to hold all chat messages
    const [messages, setMessages] = useState([
        { text: "Hello! How can I assist you today?", sender: "bot" },
    ]);

    // State to hold the current message being typed by the user
    const [inputMessage, setInputMessage] = useState('');

    // State to control whether the chatbot window is open or closed
    const [isOpen, setIsOpen] = useState(false); // Start closed by default

    // Ref for the chat display area to scroll to the bottom
    const chatDisplayRef = useRef(null);

    // Function to scroll to the bottom of the chat display whenever messages change
    useEffect(() => {
        if (chatDisplayRef.current) {
            chatDisplayRef.current.scrollTop = chatDisplayRef.current.scrollHeight;
        }
    }, [messages]);

    // Function to toggle the chatbot's open/closed state
    const toggleChatbot = () => {
        setIsOpen(!isOpen);
    };

    // Function to handle sending a message
    const handleSendMessage = async () => {
    if (inputMessage.trim() === '') return; // Don't send empty messages

    const newMessage = { text: inputMessage, sender: "user" };
    // Update messages array immediately with the new user message
    setMessages((prevMessages) => {
        const updatedMessages = [...prevMessages, newMessage];
        setInputMessage(''); // Clear input field here to ensure it clears after state update
        return updatedMessages;
    });

    try {
        // IMPORTANT: Your Python backend (Flask app.py) MUST BE RUNNING SIMULTANEOUSLY
        // on http://localhost:5000/ for this to work.
        const response = await fetch('http://localhost:5000/api/chat', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            // Send the current message AND the entire conversation history
            body: JSON.stringify({
                message: newMessage.text,
                history: messages // Send the current state of messages (which includes previous turns)
            }),
        });

        if (!response.ok) {
            const errorData = await response.json().catch(() => ({}));
            const errorMessage = errorData.error || `HTTP error! status: ${response.status}`;
            throw new Error(errorMessage);
        }

        const data = await response.json();
        const botResponseText = data.aiResponse || "I didn't get a clear response.";
        const botResponse = { text: botResponseText, sender: "bot" };
        // Update messages with the bot's response
        setMessages((prevMessages) => [...prevMessages, botResponse]);

    } catch (error) {
        console.error("Error sending message to backend:", error);
        setMessages((prevMessages) => [
            ...prevMessages,
            { text: `Oops! Could not connect to the AI. Error: ${error.message}`, sender: "bot" },
        ]);
    }
};

    // Function to handle Enter key press
    const handleKeyPress = (e) => {
        if (e.key === 'Enter') {
            handleSendMessage();
        }
    };

    return (
        // Outermost container for the chatbot. It's fixed to the bottom-right of the viewport.
        <div className="fixed bottom-4 right-4 z-50">
            {/* Chatbot Toggle Button - visible only when the chatbot window is closed */}
            {!isOpen && (
                <button
                    className="bg-blue-600 hover:bg-blue-700 text-white p-4 rounded-full shadow-lg transition-all duration-300 ease-in-out transform hover:scale-110"
                    onClick={toggleChatbot}
                    aria-label="Open Chatbot"
                >
                    {/* Simple chat icon (SVG) */}
                    <svg
                        xmlns="http://www.w3.org/2000/svg"
                        fill="none"
                        viewBox="0 0 24 24"
                        strokeWidth={1.5}
                        stroke="currentColor"
                        className="w-6 h-6"
                    >
                        <path
                            strokeLinecap="round"
                            strokeLinejoin="round"
                            d="M8.625 12a.375.375 0 1 1-.75 0 .375.375 0 0 1 .75 0Zm0 0H8.25m4.125 0a.375.375 0 1 1-.75 0 .375.375 0 0 1 .75 0Zm0 0H12m4.125 0a.375.375 0 1 1-.75 0 .375.375 0 0 1 .75 0Zm0 0h-.375M21 12c0 1.011-.68 1.916-1.666 2.373a4.522 4.522 0 0 0-.253 1.037c-.366 1.444-1.368 2.768-2.551 3.972s-2.607 2.309-3.972 2.551a4.522 4.522 0 0 0-1.037.253C10.084 21.32 9.18 22 8.169 22c-.785 0-1.488-.352-1.968-.9L3 18.25l-.018-.003A3.75 3.75 0 0 1 3 15.688V8.169c0-1.011.68-1.916 1.666-2.373a4.522 4.522 0 0 0 .253-1.037C5.285 3.315 6.287 1.991 7.47 .787s2.607-2.309 3.972-2.551a4.522 4.522 0 0 0 1.037-.253C13.916.68 14.82 0 15.831 0c.785 0 1.488.352 1.968.9L21 4.75l.018.003A3.75 3.75 0 0 1 21 7.312v8.52ZM12 9.75a2.25 2.25 0 1 0 0 4.5 2.25 2.25 0 0 0 0-4.5Z"
                        />
                    </svg>
                </button>
            )}

            {/* Chatbot Window - visible only when isOpen is true, with transition effects */}
            {isOpen && (
                <div className="max-w-md bg-white dark:bg-zinc-800 shadow-lg rounded-lg overflow-hidden
                            transform transition-all duration-500 ease-in-out
                            origin-bottom-right
                            scale-100 opacity-100
                            ">
                    <div className="flex flex-col h-[400px]">
                        {/* Chatbot Header */}
                        <div className="px-4 py-3 border-b dark:border-zinc-700 flex justify-between items-center">
                            <h2 className="text-lg font-semibold text-zinc-800 dark:text-white">
                                Chatbot Assistant
                            </h2>
                            {/* Close Button ('X' icon) */}
                            <button
                                className="text-zinc-500 hover:text-zinc-700 dark:text-zinc-400 dark:hover:text-zinc-200"
                                onClick={toggleChatbot}
                                aria-label="Close Chatbot"
                            >
                                <svg
                                    xmlns="http://www.w3.org/2000/svg"
                                    fill="none"
                                    viewBox="0 0 24 24"
                                    strokeWidth={2}
                                    stroke="currentColor"
                                    className="w-6 h-6"
                                >
                                    <path strokeLinecap="round" strokeLinejoin="round" d="M6 18L18 6M6 6l12 12" />
                                </svg>
                            </button>
                        </div>

                        {/* Chat Display Area */}
                        <div
                            ref={chatDisplayRef}
                            className="flex-1 p-3 overflow-y-auto flex flex-col space-y-2"
                            id="chatDisplay"
                        >
                            {messages.map((msg, index) => (
                                <div
                                    key={index} // Reminder: In a real app, use a unique ID for messages, not array index
                                    className={`chat-message ${
                                        msg.sender === 'user'
                                            ? 'self-end bg-blue-500' // User message style
                                            : 'self-start bg-zinc-500' // Bot message style
                                    } text-white max-w-xs rounded-lg px-3 py-1.5 text-sm`}
                                >
                                    {msg.text}
                                </div>
                            ))}
                        </div>

                        {/* Chat Input Area */}
                        <div className="px-3 py-2 border-t dark:border-zinc-700">
                            <div className="flex gap-2">
                                <input
                                    placeholder="Type your message..."
                                    className="flex-1 p-2 border rounded-lg dark:bg-zinc-700 dark:text-white dark:border-zinc-600 text-sm"
                                    id="chatInput"
                                    type="text"
                                    value={inputMessage}
                                    onChange={(e) => setInputMessage(e.target.value)}
                                    onKeyPress={handleKeyPress}
                                />
                                <button
                                    className="bg-blue-500 hover:bg-blue-700 text-white font-bold py-1.5 px-3 rounded-lg transition duration-300 ease-in-out text-sm"
                                    id="sendButton"
                                    onClick={handleSendMessage}
                                >
                                    Send
                                </button>
                            </div>
                        </div>
                    </div>
                </div>
            )}
        </div>
    );
};

export default Chatbot;
